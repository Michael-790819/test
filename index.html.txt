<script>
    // 1. 全量逐年精算數據字典 (根據您上傳的 PDF 精確錄入)
    const DB = {
        FI5: { M: {0:193, 30:321, 65:816}, F: {0:174, 30:290, 65:607} },
        RQ1: { 
            M_New: {0:3, 30:7, 46:46}, M_Ren: {0:37, 30:41, 46:94},
            F_New: {0:3, 30:17, 46:75}, F_Ren: {0:38, 30:48, 46:115}
        },
        RN1: { M: {0:35.7, 30:31.4, 46:147.8}, F: {0:39.5, 30:38.6, 46:147.3} },
        XCD: { M: {0:220, 30:276, 45:1154, 65:5002}, F: {0:235, 30:682, 45:1904, 65:3643} },
        CJ2: {
            M_New: {0:2.2, 30:9.3, 41:35.7}, M_Ren: {0:5.4, 30:11.0, 41:42.6},
            F_New: {0:2.8, 30:14.5, 41:44.4}, F_Ren: {0:5.7, 30:17.2, 41:53.3}
        }
    };

    // 2. 精算計算引擎
    function getYearlyRate(prod, gender, age, isRen = false) {
        const genderCode = gender === 'female' ? 'F' : 'M';
        let key = genderCode;
        if (prod === 'RQ1' || prod === 'CJ2') key = genderCode + (isRen ? '_Ren' : '_New');
        
        const table = DB[prod][key] || DB[prod][genderCode];
        const lookupAge = (prod === 'CJ2') ? Math.floor(age / 5) * 5 : age;

        const sortedAges = Object.keys(table).map(Number).sort((a,b)=>a-b);
        let low = sortedAges[0], high = sortedAges[sortedAges.length-1];
        for(let a of sortedAges) { if(a <= lookupAge) low = a; if(a >= lookupAge) { high = a; break; } }
        
        if (low === high) return table[low];
        return table[low] + (table[high] - table[low]) * (lookupAge - low) / (high - low);
    }

    let chartInstance = null;

    document.getElementById('insuranceForm').addEventListener('submit', function(e) {
        e.preventDefault();
        const startAge = parseInt(document.getElementById('age').value);
        const gender = document.querySelector('input[name="gender"]:checked').value;
        
        const config = [
            {id:'FI5', name:'傳富新終身壽險', amt: parseFloat(document.getElementById('fi5Amt').value)||0, factor:1, mode:'fixed', term:20},
            {id:'RQ1', name:'愛吉時一年定期癌症', amt: parseFloat(document.getElementById('rq1Amt').value)||0, factor:0.1, mode:'annual'},
            {id:'RN1', name:'保安康重大傷病', amt: parseFloat(document.getElementById('rn1Amt').value)||0, factor:1, mode:'annual'},
            {id:'XCD', name:'一年定期癌症(單位計)', amt: parseFloat(document.getElementById('xcdAmt').value)||0, factor:1, mode:'annual'},
            {id:'CJ2', name:'愛家守護五年癌症', amt: parseFloat(document.getElementById('cj2Amt').value)||0, factor:1, mode:'5year'}
        ];

        let gt1 = 0, gt2 = 0, gt80 = 0, tableHtml = "";

        config.forEach(p => {
            if(p.amt > 0) {
                const c1 = Math.round(getYearlyRate(p.id, gender, startAge, false) * p.factor * p.amt);
                const c2 = (p.mode === 'fixed') ? c1 : Math.round(getYearlyRate(p.id, gender, startAge + 1, true) * p.factor * p.amt);
                const c5 = (p.mode === 'fixed' && 5 >= p.term) ? 0 : Math.round(getYearlyRate(p.id, gender, startAge + 5, true) * p.factor * p.amt);

                let pSum = 0;
                for(let i = startAge; i <= 80; i++) {
                    const isRenewal = (i > startAge);
                    if(p.mode === 'fixed') {
                        pSum += (i < startAge + p.term) ? (getYearlyRate(p.id, gender, startAge, false) * p.amt) : 0;
                    } else {
                        pSum += getYearlyRate(p.id, gender, i, isRenewal) * p.factor * p.amt;
                    }
                }

                tableHtml += `<tr>
                    <td class="p-4 font-bold text-blue-900">${p.name}</td>
                    <td class="p-4 text-right">${p.amt} ${p.id==='XCD'?'單':'萬'}</td>
                    <td class="p-4 text-right font-bold text-blue-600">$${c1.toLocaleString()}</td>
                    <td class="p-4 text-right text-orange-600 font-semibold">$${c5.toLocaleString()}</td>
                </tr>`;
                
                gt1 += c1; gt2 += c2; gt80 += pSum;
            }
        });

        document.getElementById('premiumTableBody').innerHTML = tableHtml;
        document.getElementById('firstYearTotal').innerText = `$${Math.round(gt1).toLocaleString()}`;
        document.getElementById('secondYearTotal').innerText = `$${Math.round(gt2).toLocaleString()}`;
        document.getElementById('totalTo80').innerText = `$${Math.round(gt80).toLocaleString()}`;

        // 趨勢圖
        const labels = [];
        const totalLine = [];
        for(let i = startAge; i <= 80; i++) {
            labels.push(i + "歲");
            let yearSum = 0;
            config.forEach(p => {
                if(p.amt > 0) {
                    const isRenewal = (i > startAge);
                    if(p.mode === 'fixed') {
                        yearSum += (i < startAge + p.term) ? (getYearlyRate(p.id, gender, startAge, false) * p.amt) : 0;
                    } else {
                        yearSum += getYearlyRate(p.id, gender, i, isRenewal) * p.factor * p.amt;
                    }
                }
            });
            totalLine.push(Math.round(yearSum));
        }

        if(chartInstance) chartInstance.destroy();
        const ctx = document.getElementById('premiumChart').getContext('2d');
        chartInstance = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [{
                    label: '年度預算支出',
                    data: totalLine,
                    borderColor: '#ffffff',
                    borderWidth: 4,
                    backgroundColor: 'rgba(30, 60, 114, 0.4)',
                    fill: true,
                    tension: 0.1,
                    pointRadius: 0
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: { legend: { display: false } },
                scales: { 
                    y: { grid: { color: 'rgba(255,255,255,0.1)' }, ticks: { color: '#444' } },
                    x: { grid: { display: false }, ticks: { color: '#444' } }
                }
            }
        });

        document.getElementById('resultSection').classList.add('active');
        document.getElementById('resultSection').scrollIntoView({ behavior: 'smooth' });
    });
</script>
